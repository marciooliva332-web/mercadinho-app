function doGet() { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetFornecedores = ss.getSheetByName("Fornecedores"); var dadosFornecedores = sheetFornecedores.getDataRange().getValues(); var fornecedores = []; for (var i = 1; i < dadosFornecedores.length; i++) { if (dadosFornecedores[i][0] || dadosFornecedores[i][1]) { fornecedores.push({ id: dadosFornecedores[i][0], nome: dadosFornecedores[i][1] }); } } var sheetProdutos = ss.getSheetByName("Produtos"); var dadosProdutos = sheetProdutos.getDataRange().getValues(); var produtos = []; for (var j = 1; j < dadosProdutos.length; j++) { if (dadosProdutos[j][0] || dadosProdutos[j][1]) { produtos.push({ id: dadosProdutos[j][0], descricao: dadosProdutos[j][1] }); } } return ContentService.createTextOutput(JSON.stringify({ proveedores: fornecedores, produtos: produtos })) .setMimeType(ContentService.MimeType.JSON); } catch (erro) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: String(erro) })) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { var params = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); if (params.acao === "cadastrarFornecedor") { var sheetForn = ss.getSheetByName("Fornecedores"); var dados = sheetForn.getDataRange().getValues(); var ultimoId = dados.length > 1 ? dados[dados.length - 1][0] : 0; var novoId = Number(ultimoId) + 1; sheetForn.appendRow([novoId, params.nome]); return ContentService.createTextOutput(JSON.stringify({ status: "success", fornecedor: { id: novoId, nome: params.nome } })).setMimeType(ContentService.MimeType.JSON); } var sheetEntradas = ss.getSheetByName("Entrada de material"); var codEntrada = "NF-" + new Date().getTime(); var dataHora = new Date(); for (var i = 0; i < params.itens.length; i++) { var item = params.itens[i]; sheetEntradas.appendRow([ dataHora, codEntrada, item.codProduto, params.codFornecedor, Number(item.quantidade), Number(item.precoUnitario), Number(item.precoTotal) ]); }
